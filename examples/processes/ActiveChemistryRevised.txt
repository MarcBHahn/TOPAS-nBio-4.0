#
# This example display the track of chemistry species.
# The tracks can be accesed by the scorer manager
#
Ph/Default/Modules = 2 "g4em-dna" "TsEmDNAChemistry"

d:Ge/World/HLX= 16 nm
d:Ge/World/HLY= 16 nm
d:Ge/World/HLZ= 81 nm
s:Ge/World/Material ="G4_WATER"
b:Ge/World/Invisible = "True"

s:Ge/Segment/Parent = "World"
s:Ge/Segment/Type   = "TsCylinder"
s:Ge/Segment/Material = "G4_WATER"
s:Ge/Segment/Color    = "magenta"
d:Ge/Segment/HL       = 80.5 nm
d:Ge/Segment/RMax     = 15.0 nm

# Set the parameters to configure chemistry from TOPASChemistry.txt
includeFile = TOPASChemistry.txt   # Alternatively TOPASChemistry_Extended.txt can be used.
s:Ch/ChemistryName = "TOPASChemistry"
# This gives access of the hits with the scoring manager
b:Ch/TOPASChemistry/ChemicalStageTransportActive = "True"
# Chemical stage will finish at 100 ns
d:Ch/TOPASChemistry/ChemicalStageTimeEnd = 100 ns

# Visualization
s:Gr/view/Type = "OpenGl"
s:Gr/view/ColorBy = "ParticleType"
i:Gr/view/WindowSizeX = 900
i:Gr/view/WindowSizeY = 700 
d:Gr/view/Theta = 45 deg
d:Gr/view/Phi = 30 deg
b:Gr/view/UseSmoothTrajectories = "False"
# Color per chemical specie. Use full name of the molecules
sv:Gr/view/ColorByParticleTypeNames = 12 "e-" "proton" "alpha" "Hydroxil" "SolvatedElectron" "HydrogenPeroxide" "Hydrogen" "DiHydrogen" "Oxygen" "SuperoxideAnion" "Hydroperoxy" "Dioxidanide"
sv:Gr/view/ColorByParticleTypeColors = 12 "red" "green" "green" "magenta" "yellow" "blue" "green" "white" "red" "purple" "green" "blue"

# Source
d:So/Demo/BeamEnergy = 1 MeV
u:So/Demo/BeamEnergySpread = 0
s:So/Demo/BeamParticle = "proton"
s:So/Demo/BeamPositionDistribution = "None"
i:So/Demo/NumberOfHistoriesInRun = 1

d:Ge/BeamPosition/TransZ = -1.0 * Ge/Segment/HL nm 
d:Ge/BeamPosition/RotX = 0 deg


i:Ts/ChemistryVerbosity = 1

i:Ts/ShowHistoryCountAtInterval = 1 
i:Ts/NumberOfThreads = 1
b:Ts/ShowCPUTime = "True"
b:Ts/PauseBeforeQuit = "False"
i:Ts/Seed = 1
